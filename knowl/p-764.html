\begin{align*} \neg P&\equiv\neg \neg (p \wedge q)\\ &\equiv p \wedge q && \text{selon } \knowl{./knowl/li-oplogcomp.html}{\text{2.2.5:4}}\text{.} \end{align*}

\begin{align*} \neg Q&\equiv\neg(r\vee q)\\ &\equiv\neg r\wedge \neg q& & \text{selon } \knowl{./knowl/li-oplogDeMorgan.html}{\text{2.2.5:9}}\text{.} \end{align*}
