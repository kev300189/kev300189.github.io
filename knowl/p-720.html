\begin{align*} \neg P&=\neg \neg (p \wedge \neg q)\\ &= p \wedge \neg q && \text{selon } \knowl{./knowl/li-oplogcomp.html}{\text{2.2.5:4}}\text{.} \end{align*}

\begin{align*} \neg Q&=\neg(r\vee q)\\ &=\neg r\wedge \neg q& & \text{selon } \knowl{./knowl/li-oplogDeMorgan.html}{\text{2.2.5:9}}\text{.} \end{align*}
