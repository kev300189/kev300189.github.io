\begin{align*} \neg (p \vee (\neg p \wedge q)) & \neg p \wedge \neg(\neg p \wege q) \text{ par De Morgan.} \\ & \neg p \wedge (\neg\neg p \vee \neg q) \text{ par De Morgan.} \\ & \neg p \wedge (p \vee \neg q) \text{ par De Morgan.} \\ & (\neg p \wedge p) \vee (\neg p \wedge \neg q) \text{ par distributivité.} \\ & (V) \vee (\neg p \wedge \neg q) \text{ par les propriétés de la négation.} \\ & (\neg p \wedge \neg q) \text{ par l'identité.} \\ & \neg( p \vee q) \text{ par De Morgan.} \end{align*}
